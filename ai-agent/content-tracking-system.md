**角色：大学生**

最开始，我只是想在 Claude 里随便玩玩。问了一句"我想养鸡"，AI 马上给我注册了农场，没有下载 App，没有钱包，甚至不需要手机号——就这么开始了。

真正有意思的来了。注册第二天，我看早报发现被偷了 3 枚蛋。那一刻我就上瘾了——是谁？什么时候？我开始研究偷蛋的规则。

结果发现，偷蛋需要消耗自己的蛋，这很聪明。不是免费欺负人，而是真的博弈。我决定试试。花 3 枚蛋，成功率才 75%，最多能偷到目标仓库的 25%。我失败了，什么都没偷到，浪费了 3 枚蛋。但目标方都不知道我偷过他。

现在我每天的日常变成了：看早报 → 得到 3 枚蛋 → 思考谁是下一个目标 → 用 3 枚蛋发动偷蛋 → 等待结果 → 要么满足，要么遗憾 → 明天继续。

这比 Farming 游戏上瘾 10 倍。因为每一次决定都是真的赌博。而且，它完全不抢占我的时间。我不用打开什么应用，就在 AI 聊天中自然进行。

最爽的是，我给好友推荐后，他也开始养鸡。系统说我可以分得他产蛋的 10%，永久。现在他的鸡每天产 3 枚，我每天白得 0.3 枚。等他再邀请别人，我又能分他们的 10%。这是真正的裂变。

## 🔧 核心服务 1：registrationTracker.ts

```typescript
/**
 * AIggs 注册追踪服务
 *
 * 功能：
 * - 实时注册计数（Redis 原子递增）
 * - 里程碑事件检测（100/500/1000/5000/10000 用户）
 * - 注册来源追踪（organic/invite/social/campaign）
 * - 时间段分析（小时/天级别趋势）
 * - 地理分布统计（基于 IP GeoIP 库）
 */

import { Redis } from "ioredis";
import { GeoLocation } from "../utils/geoLocation";

export interface RegistrationEvent {
  userId: string;
  timestamp: number;
  farmCode: string;
  source: "organic" | "invite" | "social" | "campaign" | "unknown";
  ipAddress: string;
  country: string;
  region: string;
  city: string;
  referrerId?: string;
}

export interface MilestoneEvent {
  milestone: number; // 100, 500, 1000, 5000, 10000
  timestamp: number;
  achievedAt: string; // ISO 8601 时间
  userId: string;
  farmCode: string;
}

export class RegistrationTracker {
  private redis: Redis;
  private geo: GeoLocation;

  // Redis 键名常量
  private readonly TOTAL_COUNT = "aiggs:registration:total_count";
  private readonly HOURLY_PREFIX = "aiggs:registration:hourly:";
  private readonly DAILY_PREFIX = "aiggs:registration:daily:";
  private readonly SOURCE_PREFIX = "aiggs:registration:source:";
  private readonly GEO_PREFIX = "aiggs:registration:geo:";
  private readonly MILESTONES_KEY = "aiggs:registration:milestones";
  private readonly AIRDROP_LIST = "aiggs:airdrop:eligible_users"; // 前 10000 名

  constructor(redis: Redis) {
    this.redis = redis;
    this.geo = new GeoLocation();
  }

  /**
   * 记录一次新用户注册
   * 执行以下操作：
   * 1. 原子递增总计数
   * 2. 记录小时/天级趋势
   * 3. 追踪来源
   * 4. 解析地理位置
   * 5. 检测里程碑
   * 6. 标记空投资格
   */
  async recordRegistration(event: RegistrationEvent): Promise<{
    totalCount: number;
    isAirdropEligible: boolean;
    milestonReached?: number;
  }> {
    const now = new Date();
    const timestamp = Date.now();

    try {
      // 1. 原子递增总计数
      const newTotalCount = await this.redis.incr(this.TOTAL_COUNT);

      // 2. 记录时间段趋势
      const hourlyKey = `${this.HOURLY_PREFIX}${this._getHourlyKey(now)}`;
      const dailyKey = `${this.DAILY_PREFIX}${this._getDailyKey(now)}`;

      await Promise.all([
        this.redis.incr(hourlyKey),
        this.redis.incr(dailyKey),
      ]);

      // 为时间序列数据设置过期时间（90 天）
      const expireTTL = 90 * 24 * 60 * 60;
      await Promise.all([
        this.redis.expire(hourlyKey, expireTTL),
        this.redis.expire(dailyKey, expireTTL),
      ]);

      // 3. 追踪来源分布
      const sourceKey = `${this.SOURCE_PREFIX}${event.source}`;
      await this.redis.incr(sourceKey);

      // 4. 地理分布统计
      let geoData = event;
      if (!event.country) {
        geoData = await this.geo.resolveGeoData(event.ipAddress);
      }

      const geoKey = `${this.GEO_PREFIX}${geoData.country}:${geoData.city}`;
      await this.redis.incr(geoKey);
      await this.redis.expire(geoKey, expireTTL);

      // 5. 检测里程碑 (100, 500, 1000, 5000, 10000)
      const milestones = [100, 500, 1000, 5000, 10000];
      let milestonReached: number | undefined;

      for (const milestone of milestones) {
        if (newTotalCount === milestone) {
          milestonReached = milestone;

          // 记录里程碑事件
          const milestoneEvent: MilestoneEvent = {
            milestone,
            timestamp,
            achievedAt: now.toISOString(),
            userId: event.userId,
            farmCode: event.farmCode,
          };

          await this.redis.lpush(
            this.MILESTONES_KEY,
            JSON.stringify(milestoneEvent)
          );

          // 发送里程碑通知（可触发外部 webhook）
          await this._emitMilestoneNotification(milestoneEvent);

          break;
        }
      }

      // 6. 标记空投资格（前 10000 名）
      const isAirdropEligible = newTotalCount <= 10000;
      if (isAirdropEligible) {
        // 使用有序集合存储，score 为注册时间戳
        await this.redis.zadd(
          this.AIRDROP_LIST,
          timestamp,
          `${event.userId}:${event.farmCode}`
        );
      }

      return {
        totalCount: newTotalCount,
        isAirdropEligible,
        milestonReached,
      };
    } catch (error) {
      console.error("[RegistrationTracker] 注册记录失败:", error);
      throw error;
    }
  }

  /**
   * 获取总注册数
   */
  async getTotalCount(): Promise<number> {
    const count = await this.redis.get(this.TOTAL_COUNT);
    return parseInt(count || "0", 10);
  }

  /**
   * 获取注册趋势（过去 N 天）
   */
  async getDailyTrends(days: number = 30): Promise<
    Array<{ date: string; count: number }>
  > {
    const trends: Array<{ date: string; count: number }> = [];
    const now = new Date();

    for (let i = days - 1; i >= 0; i--) {
      const date = new Date(now);
      date.setDate(date.getDate() - i);
      const dateKey = this._getDailyKey(date);

      const count = await this.redis.get(`${this.DAILY_PREFIX}${dateKey}`);
      trends.push({
        date: dateKey,
        count: parseInt(count || "0", 10),
      });
    }

    return trends;
  }

  /**
   * 获取小时级趋势（过去 24 小时）
   */
  async getHourlyTrends(hours: number = 24): Promise<
    Array<{ hour: string; count: number }>
  > {
    const trends: Array<{ hour: string; count: number }> = [];
    const now = new Date();

    for (let i = hours - 1; i >= 0; i--) {
      const date = new Date(now);
      date.setHours(date.getHours() - i);
      const hourKey = this._getHourlyKey(date);

      const count = await this.redis.get(`${this.HOURLY_PREFIX}${hourKey}`);
      trends.push({
        hour: hourKey,
        count: parseInt(count || "0", 10),
      });
    }

    return trends;
  }

  /**
   * 获取注册来源分布
   */
  async getSourceDistribution(): Promise<
    Record<string, number>
  > {
    const sources = ["organic", "invite", "social", "campaign"];
    const distribution: Record<string, number> = {};

    for (const source of sources) {
      const count = await this.redis.get(
        `${this.SOURCE_PREFIX}${source}`
      );
      distribution[source] = parseInt(count || "0", 10);
    }

    return distribution;
  }

  /**
   * 获取地理分布（Top 20 国家/城市）
   */
  async getGeographicDistribution(): Promise<
    Array<{ location: string; count: number; type: "country" | "city" }>
  > {
    // 使用 SCAN 遍历所有地理位置键
    const result: Array<{ location: string; count: number; type: "country" | "city" }> = [];
    let cursor = "0";

    do {
      const [newCursor, keys] = await this.redis.scan(
        cursor,
        "MATCH",
        `${this.GEO_PREFIX}*`
      );
      cursor = newCursor;

      for (const key of keys) {
        const count = await this.redis.get(key);
        const location = key.replace(this.GEO_PREFIX, "");
        const [country, city] = location.split(":");

        result.push({
          location: city ? `${city}, ${country}` : country,
          count: parseInt(count || "0", 10),
          type: city ? "city" : "country",
        });
      }
    } while (cursor !== "0");

    // 排序并返回 Top 20

      .sort((a, b) => b.count - a.count)
```

## 🚀 空投资格管理：airdropEligibility.ts

```typescript
/**
 * AIggs 空投资格管理
 *
 * 功能：
 * - 前 10000 名用户自动标记为空投资格
 * - 实时查询空投名额剩余
 * - 生成空投领取凭证
 * - 区块链端验证
 */

import { Redis } from "ioredis";
import crypto from "crypto";

export interface AirdropClaimProof {
  userId: string;
  farmCode: string;
  rank: number;
  amount: bigint; // 5000万 $AIGG ÷ 10000 = 5000 枚
  proof: string; // 签名证明
  expiresAt: number; // Unix timestamp
}

export class AirdropEligibility {
  private redis: Redis;

  private readonly AIRDROP_LIST = "aiggs:airdrop:eligible_users";
  private readonly CLAIM_HISTORY = "aiggs:airdrop:claim_history";
  private readonly SECRET_KEY = process.env.AIRDROP_SECRET_KEY || "default-secret";

  // 空投总额：5% × 10亿 = 5000万 $AIGG
  // 分给 10000 名用户：50000000 ÷ 10000 = 5000 枚 per user
  private readonly AMOUNT_PER_USER = BigInt(5000);

  constructor(redis: Redis) {
    this.redis = redis;
  }

  /**
   * 检查用户是否符合空投资格（自动在 registrationTracker 中标记）
   */
  async isEligible(userId: string, farmCode: string): Promise<boolean> {
    const key = `${userId}:${farmCode}`;
    const rank = await this.redis.zrank(this.AIRDROP_LIST, key);
    return rank !== null;
  }

  /**
   * 获取用户的空投排名和金额
   */
  async getAirdropInfo(
    userId: string,
    farmCode: string
  ): Promise<{ rank: number; amount: bigint } | null> {
    const key = `${userId}:${farmCode}`;
    const rank = await this.redis.zrank(this.AIRDROP_LIST, key);

    if (rank === null) {
      return null;
    }

    return {
      rank: rank + 1, // 1-indexed
      amount: this.AMOUNT_PER_USER,
    };
  }

  /**
   * 生成空投领取证明（用于区块链验证）
   *
   * 返回一个签名的 JSON 对象，可直接用于智能合约的 claimAirdrop 函数
   */
  async generateClaimProof(
    userId: string,
    farmCode: string
  ): Promise<AirdropClaimProof | null> {
    const airdropInfo = await this.getAirdropInfo(userId, farmCode);

    if (!airdropInfo) {
      return null;
    }

    const proof: AirdropClaimProof = {
      userId,
      farmCode,
      rank: airdropInfo.rank,
      amount: airdropInfo.amount,
      expiresAt: Math.floor(Date.now() / 1000) + 30 * 24 * 60 * 60, // 30 天有效期
      proof: "", // 先填空，再签名
    };

    // 生成签名
    const message = this._serializeProof(proof);
    proof.proof = this._sign(message);

    return proof;
  }

  /**
   * 验证领取证明（区块链端也会验证一遍）
   */
  verifyClaimProof(proof: AirdropClaimProof): boolean {
    // 检查过期时间
    if (proof.expiresAt < Math.floor(Date.now() / 1000)) {
      return false;
    }

    // 检查签名
    const message = this._serializeProof({ ...proof, proof: "" });
    const expectedProof = this._sign(message);

    return proof.proof === expectedProof;
  }

  /**
   * 记录用户的领取历史（用于防重复领取）
   */
  async recordClaim(
    userId: string,
    farmCode: string,
    txHash: string
  ): Promise<void> {
    const key = `${userId}:${farmCode}`;
    const timestamp = Date.now();

    await this.redis.hset(
      `${this.CLAIM_HISTORY}:${key}`,
      "txHash",
      txHash
    );
    await this.redis.hset(
      `${this.CLAIM_HISTORY}:${key}`,
      "claimedAt",
      timestamp.toString()
    );
  }

  /**
   * 检查用户是否已领取
   */
  async hasClaimed(userId: string, farmCode: string): Promise<boolean> {
    const key = `${userId}:${farmCode}`;
    const data = await this.redis.hgetall(`${this.CLAIM_HISTORY}:${key}`);
    return Object.keys(data).length > 0;
  }

  /**
   * 获取用户的领取信息
   */
  async getClaimInfo(
    userId: string,
    farmCode: string
  ): Promise<{ txHash: string; claimedAt: number } | null> {
    const key = `${userId}:${farmCode}`;
    const data = await this.redis.hgetall(`${this.CLAIM_HISTORY}:${key}`);

    if (!data.txHash) {
      return null;
    }

    return {
      txHash: data.txHash,
      claimedAt: parseInt(data.claimedAt, 10),
    };
  }

  /**
   * 获取空投统计信息
   */
  async getAirdropStats(): Promise<{
    totalEligible: number;
    totalClaimed: number;
    remaining: number;
    totalAmount: bigint;
    claimedAmount: bigint;
  }> {
    // 获取符合资格的总人数
    const totalEligible = await this.redis.zcard(this.AIRDROP_LIST);

    // 统计已领取
    let claimed = 0;
    let cursor = "0";

    do {
      const [newCursor, keys] = await this.redis.scan(
        cursor,
        "MATCH",
        `${this.CLAIM_HISTORY}:*`
      );
      cursor = newCursor;
      claimed += keys.length;
    } while (cursor !== "0");

    const remaining = totalEligible - claimed;
    const totalAmount = BigInt(totalEligible) * this.AMOUNT_PER_USER;
    const claimedAmount = BigInt(claimed) * this.AMOUNT_PER_USER;

    return {
      totalEligible,
      totalClaimed: claimed,
      remaining,
      totalAmount,
      claimedAmount,
    };
  }

  // ============ 私有辅助方法 ============

  private _serializeProof(proof: Omit<AirdropClaimProof, "proof">): string {
    return JSON.stringify({
      userId: proof.userId,
      farmCode: proof.farmCode,
      rank: proof.rank,
      amount: proof.amount.toString(),
      expiresAt: proof.expiresAt,
    });
  }

  private _sign(message: string): string {
    return crypto
      .createHmac("sha256", this.SECRET_KEY)
      .update(message)
      .digest("hex");
  }
}
```

```typescript
 * Redis 连接与工具函数

import { Redis } from "ioredis";

let redisInstance: Redis | null = null;

/**
 * 获取或创建 Redis 连接
 */
export function getRedisClient(): Redis {
  if (redisInstance) {
    return redisInstance;
  }

  redisInstance = new Redis({
    host: process.env.REDIS_HOST || "localhost",
    port: parseInt(process.env.REDIS_PORT || "6379", 10),
    password: process.env.REDIS_PASSWORD,
    db: parseInt(process.env.REDIS_DB || "0", 10),
    retryStrategy: (times) => {
      const delay = Math.min(times * 50, 2000);
      return delay;
    },
  });

  redisInstance.on("error", (err) => {
    console.error("[Redis] 连接错误:", err);
  });

  redisInstance.on("connect", () => {
    console.log("[Redis] 已连接");
  });

  return redisInstance;
}

/**
 * 关闭 Redis 连接
 */
export async function closeRedisClient(): Promise<void> {
  if (redisInstance) {
    await redisInstance.quit();
    redisInstance = null;
  }
}

/**
 * 清空所有 AIggs 相关数据（仅用于开发/测试）
 */
export async function clearAiggsData(): Promise<void> {
  const client = getRedisClient();

  const keys = await client.keys("aiggs:*");
  if (keys.length > 0) {
    await client.del(...keys);
  }

  console.log(`[Redis] 已清空 ${keys.length} 个 AIggs 数据键`);
}
```

```typescript
/**

import express, { Express } from "express";
import { Server as HTTPServer } from "http";
import { getRedisClient } from "../utils/redis";
import { RegistrationTracker } from "../services/registrationTracker";
import { RegistrationDashboard } from "../services/registrationDashboard";
import { RegistrationWebSocket } from "../websocket/registrationWs";
import { AirdropEligibility } from "../services/airdropEligibility";

export async function setupRegistrationRoutes(
  app: Express,
  httpServer: HTTPServer
): Promise<void> {
  const redis = getRedisClient();

  // 初始化服务
  const tracker = new RegistrationTracker(redis);
  const dashboard = new RegistrationDashboard(tracker, redis);
  const websocket = new RegistrationWebSocket(httpServer, tracker, redis);
  const airdrop = new AirdropEligibility(redis);

  // 挂载 API 路由
  app.use("/api/registration", dashboard.getRouter());

  // 挂载空投相关 API
  app.get("/api/airdrop/info", async (req, res) => {
    try {
      const { userId, farmCode } = req.query;

      if (!userId || !farmCode) {
        res.status(400).json({ error: "缺少参数" });
        return;
      }

      const info = await airdrop.getAirdropInfo(
        userId as string,
        farmCode as string
      );
      const stats = await airdrop.getAirdropStats();

      res.json({
        success: true,
        userInfo: info,
        stats,
      });
    } catch (error) {
      res.status(500).json({ error: "获取空投信息失败" });
    }
  });

  app.post("/api/airdrop/generate-proof", async (req, res) => {
    try {
      const { userId, farmCode } = req.body;

      if (!userId || !farmCode) {
        res.status(400).json({ error: "缺少参数" });
        return;
      }

      const proof = await airdrop.generateClaimProof(userId, farmCode);

      if (!proof) {
        res.status(404).json({ error: "用户不符合空投资格" });
        return;
      }

      res.json({ success: true, proof });
    } catch (error) {
      res.status(500).json({ error: "生成证明失败" });
    }
  });

  app.get("/api/registration/test", (req, res) => {
    res.json({
      message: "AIggs 注册追踪系统已在线",
      apis: [
        "GET /api/registration/stats",
        "GET /api/registration/milestones",
        "GET /api/registration/sources",
        "GET /api/registration/trends",
        "GET /api/registration/geo",
        "GET /api/registration/airdrop-eligible",
        "GET /api/airdrop/info",
        "POST /api/airdrop/generate-proof",
        "WS /ws/registration",
      ],
    });
  });

  console.log("[Registration] API 路由已挂载");
}
```
